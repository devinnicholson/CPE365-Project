create table country (
name varchar(28),
id int autoincrement primary key
);

create table university (
name varchar(60),
id int autoincrement primary key,
country int references country
);

create table ranking (
rank int,
university int references university,
teaching float,
international float,
research float,
citations float,
income float,
total float,
nStudents int,
studentFacultyRatio float,
pctInternational int,
femaleMaleRatio varchar(12),
year int,
primary key(university, year)
);

create table expenditure (
country int references country,
type varchar(7),
expenditure float,
primary key(country, type)
);

create table attainment (
country int refences country,
attainment float,
gender varchar(6),
degree varchar(8),
primary key(country, gender, degree)
);
